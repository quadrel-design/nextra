import fs from 'node:fs/promises'
import { compileMdx } from 'nextra/compile'
import { MDXRemote } from 'nextra/mdx-remote'

# Remote Content

> [!NOTE]
>
> You can find information about remote docs support in
> [Nextra 3 blogpost](https://the-guild.dev/blog/nextra-3#remote-docs-support)
> announcement.

export async function Example() {
  const pageContent = await fs.readFile(
    '../examples/swr-site/app/[lang]/remote/graphql-eslint/[[...slug]]/page.tsx',
    'utf8'
  )
  const rawJs = await compileMdx(`\`\`\`jsx
${pageContent
  .replace('// @ts-expect-error -- fixme\n', '')
  .split('\n')
  .slice(1, -1)
  .join('\n')
    .replace("lang: 'en',\n    ...(route && { slug: route.split('/') })", "slug: route.split('/')")}
\`\`\``)
  return <MDXRemote compiledSource={rawJs} />
}
